\section{Phân tích \& Refactoring: Auction State Logic}

% ----------------------------------------------------------
% ISSUE #2 — Slide 1: Problem
% ----------------------------------------------------------
\begin{frame}{Issue \#2 — Vi phạm DRY: Duplicate \texttt{productStatus} Logic}
  \textbf{Vấn đề:} Khối logic xác định trạng thái auction được copy nguyên vẹn
  vào 2 route handler khác nhau.

  \bigskip
  \textbf{Nguyên tắc vi phạm:} \alert{DRY} (Don't Repeat Yourself)

  \bigskip
  \begin{block}{Business rule bị encode 2 lần}
    \begin{itemize}
      \item \texttt{GET /detail} → 5-nhánh if/else (dòng $\sim$162)
      \item \texttt{GET /complete-order} → 4-nhánh if/else (dòng $\sim$984)
      \item Hai bản đã \textbf{drift}: bản thứ hai thiếu nhánh \texttt{ACTIVE} cuối
    \end{itemize}
  \end{block}

  \bigskip
  \textbf{Rủi ro kỹ thuật:}
  \begin{itemize}
    \item Thêm state \texttt{DISPUTED} → phải sửa \textbf{cả hai chỗ}; nếu quên một, hai route trả kết quả mâu thuẫn nhau với cùng một product
    \item Không có unit nào để test — logic embedded sâu trong HTTP handler
    \item Business rule bị coupled với HTTP routing layer
  \end{itemize}
\end{frame}

% ----------------------------------------------------------
% ISSUE #2 — Slide 2: Solution
% ----------------------------------------------------------
\begin{frame}[fragile]{Issue \#2 — Refactoring: Extract \texttt{resolveAuctionStatus()}}
  \textbf{Chiến lược:} Extract Function → đưa về \texttt{services/auction/auction-state.js}

  \begin{columns}[t]
    \column{0.48\textwidth}
    \textbf{\alert{Trước}} — inline, lặp lại:
    \footnotesize
    \begin{verbatim}
// Cùng khối xuất hiện 2 lần
const now = new Date();
const end = new Date(product.end_at);
let status = 'ACTIVE';
if (product.is_sold === true)
  status = 'SOLD';
else if (...)
  status = 'CANCELLED';
// ... 3 nhánh nữa
    \end{verbatim}

    \column{0.48\textwidth}
    \textbf{\textcolor{mygreen}{Sau}} — một hàm duy nhất:
    \footnotesize
    \begin{verbatim}
// auction-state.js
export function
  resolveAuctionStatus(product) {
  if (product.is_sold === true)
    return 'SOLD';
  if (product.is_sold === false)
    return 'CANCELLED';
  // ...
  return 'ACTIVE';
}

// Trong route handler:
const status =
  resolveAuctionStatus(product);
    \end{verbatim}
  \end{columns}

  \bigskip
  \textbf{Kết quả:} Pure function, zero dependency, test được bằng \texttt{assert(resolveAuctionStatus(mockProduct) === 'PENDING')}
\end{frame}

% ----------------------------------------------------------
% ISSUE #6 — Slide 1: Problem
% ----------------------------------------------------------
\begin{frame}{Issue \#6 — Vi phạm OCP: State Taxonomy Không Tập Trung}
  \textbf{Vấn đề:} Logic phân loại state auction được viết trực tiếp trong route handlers dưới dạng if/else inline — không có module "đóng" bảo vệ core logic.

  \bigskip
  \textbf{Nguyên tắc vi phạm:} \alert{OCP} (Open/Closed Principle)

  \bigskip
  \begin{alertblock}{Phân biệt quan trọng}
    Guard condition (\texttt{if (!product) return 404}) \textbf{không} phải OCP violation.\\
    State classification (\texttt{if (...) status = 'PENDING'}) \textbf{là} OCP violation — đây là \emph{trục mở rộng domain}.
  \end{alertblock}

  \bigskip
  \textbf{Hậu quả khi thêm state \texttt{PAUSED}:}
  \begin{itemize}
    \item Phải \textbf{mở (modify)} code đang chạy trong production ở nhiều file
    \item Không có "điểm mở rộng" định nghĩa sẵn — developer phải grep toàn codebase
    \item Mỗi file có thể implement logic mở rộng theo cách khác nhau → inconsistency
  \end{itemize}
\end{frame}

% ----------------------------------------------------------
% ISSUE #6 — Slide 2: Solution
% ----------------------------------------------------------
\begin{frame}{Issue \#6 — Refactoring: Encapsulate Variation (State Module)}
  \textbf{Chiến lược:} Tập trung toàn bộ state taxonomy vào một module duy nhất.\\
  Module này \textbf{đóng} với modification, \textbf{mở} cho extension.

  \bigskip
  \begin{block}{\texttt{auction-state.js} — Single Point of Extension}
    \begin{itemize}
      \item \texttt{resolveAuctionStatus(product)} → trả về state string
      \item \texttt{canBid(status)} → guard function
      \item \texttt{canAccessOrder(status)} → guard function
      \item \texttt{requiresAccessCheck(status)} → guard function
    \end{itemize}
  \end{block}

  \bigskip
  \textbf{Thêm state \texttt{PAUSED} sau refactor:}
  \begin{enumerate}
    \item Thêm 1 điều kiện vào \texttt{resolveAuctionStatus()} trong \texttt{auction-state.js}
    \item Thêm guard function \texttt{isPaused(status)} nếu cần
    \item \textbf{Không sửa} bất kỳ route handler hay notifier nào
  \end{enumerate}

  \bigskip
  \textbf{Pattern áp dụng:} \emph{Encapsulate Variation} — mọi variation của "state" sống trong một module. Caller chỉ nhận kết quả, không biết logic phân loại.
\end{frame}

% ----------------------------------------------------------
% ISSUE #14 — Slide 1: Problem
% ----------------------------------------------------------
\begin{frame}{Issue \#14 — Vi phạm DRY: \texttt{auctionEndNotifier} Tự Tính State}
  \textbf{Vấn đề:} \texttt{auctionEndNotifier.js} — module chạy độc lập — tự encode business rule về auction state theo cách riêng, tạo ra \textbf{nguồn sự thật thứ ba}.

  \bigskip
  \textbf{Nguyên tắc vi phạm:} \alert{DRY} ở cấp \emph{cross-module knowledge duplication}

  \bigskip
  \begin{block}{Ba nguồn sự thật trong cùng một hệ thống}
    \begin{itemize}
      \item \texttt{GET /detail}: 5-nhánh if/else (kiểm tra \texttt{is\_sold} + \texttt{highest\_bidder\_id})
      \item \texttt{GET /complete-order}: 4-nhánh if/else (đã drift)
      \item \texttt{auctionEndNotifier.js}: \texttt{if (auction.highest\_bidder\_id)} — 2-nhánh, \textbf{không kiểm tra \texttt{is\_sold}}
    \end{itemize}
  \end{block}

  \bigskip
  \textbf{Bug tiềm ẩn:} Product đã SOLD vẫn có \texttt{highest\_bidder\_id} $\neq$ null.\\
  Nếu query trả về SOLD product, notifier sẽ gửi email ``Bạn đã thắng đấu giá!'' cho một giao dịch đã hoàn tất từ trước.
\end{frame}

% ----------------------------------------------------------
% ISSUE #14 — Slide 2: Solution
% ----------------------------------------------------------
\begin{frame}[fragile]{Issue \#14 — Refactoring: Notifier Depend vào \texttt{auction-state.js}}
  \textbf{Chiến lược:} Notifier không tự tính state — import và dùng \texttt{resolveAuctionStatus()}.

  \begin{columns}[t]
    \column{0.46\textwidth}
    \textbf{\alert{Trước}} — implicit, thiếu check:
    \footnotesize
    \begin{verbatim}
// implicit state check, 2 nhánh
// không check is_sold
if (auction.highest_bidder_id) {
  await sendMail(winner, ...);
  await sendMail(seller, ...);
} else {
  await sendMail(seller, ...);
}
    \end{verbatim}

    \column{0.50\textwidth}
    \textbf{\textcolor{mygreen}{Sau}} — explicit, đúng rule:
    \footnotesize
    \begin{verbatim}
import { resolveAuctionStatus }
  from '../services/auction/
         auction-state.js';

const status =
  resolveAuctionStatus(auction);

if (status === 'PENDING') {
  await notifyWinner(auction);
  await notifySeller(auction);
} else if (status === 'EXPIRED') {
  await notifySellerNoBid(auction);
}
// SOLD/CANCELLED: không gửi email
    \end{verbatim}
  \end{columns}

  \bigskip
  \textbf{Key Takeaway:}
  \begin{itemize}
    \item \textbf{Single Source of Truth}: thay đổi rule tại 1 nơi, propagate toàn hệ thống
    \item \textbf{Dependency on Abstraction}: notifier depend vào function contract, không depend vào raw DB fields
  \end{itemize}
\end{frame}

\section{Phân tích \& Refactoring: \texttt{review.model.js} API}

% ----------------------------------------------------------
% ISSUE #11 — Slide 1: Problem
% ----------------------------------------------------------
\begin{frame}{Issue \#11 — Vi phạm DRY + ISP: Dual INSERT API}
  \textbf{Vấn đề:} \texttt{review.model.js} export \textbf{hai hàm} thực hiện cùng một INSERT vào bảng \texttt{reviews}, với \textbf{field naming convention khác nhau}.

  \bigskip
  \textbf{Nguyên tắc vi phạm:} \alert{DRY} + \alert{ISP}

  \bigskip
  \begin{block}{Hai hàm, hai convention — cùng một DB operation}
    \begin{itemize}
      \item \texttt{createReview(reviewData)} → caller dùng \texttt{reviewee\_id} \quad[\textit{seller.route.js}]
      \item \texttt{create(data)} → caller dùng \texttt{reviewed\_user\_id} \quad[\textit{product/account.route.js}]
      \item Cả hai đều INSERT vào cùng column \texttt{reviewee\_id} trong DB
    \end{itemize}
  \end{block}

  \bigskip
  \textbf{Rủi ro kỹ thuật:}
  \begin{itemize}
    \item Dùng nhầm hàm → truyền sai field name → \texttt{reviewee\_id} lưu \texttt{NULL} silently
    \item Đổi DB schema (rename column) → phải sửa cả 2 hàm + tất cả caller
    \item Developer mới không có cách biết hàm nào nên dùng khi nào
  \end{itemize}
\end{frame}

% ----------------------------------------------------------
% ISSUE #11 — Slide 2: Solution
% ----------------------------------------------------------
\begin{frame}[fragile]{Issue \#11 — Refactoring: Một API duy nhất, camelCase nhất quán}
  \textbf{Chiến lược:} Merge thành một hàm canonical với destructuring params (camelCase).

  \begin{columns}[t]
    \column{0.46\textwidth}
    \textbf{\alert{Trước}} — 2 hàm, 2 convention:
    \footnotesize
    \begin{verbatim}
// seller.route.js
await reviewModel.createReview({
  reviewer_id: sellerId,
  reviewee_id: bidder_id, // ← field A
  product_id: productId, ...
});

// product.route.js
await reviewModel.create({
  reviewer_id: reviewerId,
  reviewed_user_id: revieweeId, // ← field B
  product_id: ..., ...
});
    \end{verbatim}

    \column{0.50\textwidth}
    \textbf{\textcolor{mygreen}{Sau}} — 1 hàm, camelCase:
    \footnotesize
    \begin{verbatim}
// review.model.js
export function createReview({
  reviewerId, revieweeId,
  productId, rating,
  comment = null }) {
  return db('reviews').insert({
    reviewer_id: reviewerId,
    reviewee_id: revieweeId, // ← 1 nơi duy nhất
    product_id:  productId,
    ...
  });
}
// Tất cả caller dùng cùng interface
    \end{verbatim}
  \end{columns}

  \bigskip
  \textbf{Key Takeaway:} Module là \emph{anti-corruption layer} — chỉ \texttt{review.model.js} biết \texttt{reviewee\_id} column name. Caller chỉ biết domain term \texttt{revieweeId}.
\end{frame}

% ----------------------------------------------------------
% ISSUE #12 — Slide 1: Problem
% ----------------------------------------------------------
\begin{frame}{Issue \#12 — Vi phạm DRY + YAGNI: Dual UPDATE API}
  \textbf{Vấn đề:} Hai hàm UPDATE khác nhau ở số lượng tham số WHERE — nhưng tham số thêm (\texttt{reviewee\_id}) là \textbf{redundant predicate} về mặt domain.

  \bigskip
  \textbf{Nguyên tắc vi phạm:} \alert{DRY} + \alert{YAGNI}

  \bigskip
  \begin{alertblock}{Domain invariant quan trọng}
    \textbf{(reviewer\_id, product\_id) đã đủ unique} — một người chỉ review một lần cho một sản phẩm.\\
    Thêm \texttt{reviewee\_id} vào WHERE không thu hẹp result set — nhưng nếu truyền sai, \textbf{UPDATE 0 rows silently}.
  \end{alertblock}

  \bigskip
  \textbf{Biểu hiện inconsistency trong \texttt{seller.route.js}:}
  \begin{itemize}
    \item \texttt{POST /products/:id/rate} → dùng \texttt{updateByReviewerAndProduct()} (3 params)
    \item \texttt{PUT /products/:id/rate} → dùng \texttt{updateReview()} (4 params)
    \item Cùng một file, hai route làm cùng việc → hai hàm khác nhau
  \end{itemize}
\end{frame}

% ----------------------------------------------------------
% ISSUE #12 — Slide 2: Solution
% ----------------------------------------------------------
\begin{frame}[fragile]{Issue \#12 — Refactoring: Encode Domain Invariant vào WHERE}
  \textbf{Chiến lược:} Giữ lại 2-param version, bỏ 4-param version, đổi sang camelCase.

  \begin{columns}[t]
    \column{0.46\textwidth}
    \textbf{\alert{Trước}} — 2 hàm, 2 WHERE:
    \footnotesize
    \begin{verbatim}
// 4-param (seller.route PUT /rate)
await reviewModel.updateReview(
  sellerId,
  highest_bidder_id, // thừa
  productId, data
);

// 3-param (seller.route POST /rate)
await reviewModel
  .updateByReviewerAndProduct(
    sellerId, productId, data
);
// ← Cùng file, 2 hàm khác nhau
    \end{verbatim}

    \column{0.50\textwidth}
    \textbf{\textcolor{mygreen}{Sau}} — 1 hàm, WHERE đúng:
    \footnotesize
    \begin{verbatim}
// review.model.js
export function updateReview(
  reviewerId, productId, updateData
) {
  return db('reviews')
    .where('reviewer_id', reviewerId)
    .where('product_id',  productId)
    // (reviewer, product) = unique key
    .update(updateData);
}

// Mọi caller đều dùng:
await reviewModel.updateReview(
  userId, productId, data
);
    \end{verbatim}
  \end{columns}

  \bigskip
  \textbf{Key Takeaway:} WHERE clause trở thành \emph{executable documentation} của invariant domain: ``một reviewer chỉ có một review cho một product''. Code = specification.
\end{frame}

% Summary Slide — Issues #11 & #12
% ----------------------------------------------------------
\begin{frame}{Tổng Kết Issues \#11 \& \#12 — Model Layer Anti-Pattern}
  \begin{block}{Root Cause chung}
    \textbf{Incremental development} không có consolidation: mỗi lần thêm feature mới, một hàm mới được tạo thay vì mở rộng hàm hiện có → duplicate API tích lũy.
  \end{block}

  \bigskip
  \begin{columns}[t]
    \column{0.48\textwidth}
    \textbf{Triệu chứng:}
    \begin{itemize}
      \item 2 INSERT functions: \texttt{create()} vs \texttt{createReview()}
      \item 2 UPDATE functions: 3-param vs 4-param
      \item Caller buộc phải biết \emph{convention nội bộ} của model
      \item Silent data loss khi truyền sai field name
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Kết quả sau refactor:}
    \begin{itemize}
      \item 1 INSERT: \texttt{createReview(\{...camelCase\})}
      \item 1 UPDATE: \texttt{updateReview(userId, productId, data)}
      \item Domain invariant encoded trong WHERE clause
      \item 3 call-site files updated atomically
    \end{itemize}
  \end{columns}

  \bigskip
  \begin{center}
    \textit{Model là \textbf{anti-corruption layer} — caller không cần biết DB column naming convention. Chỉ có model mới được biết.}
  \end{center}
\end{frame}

\section{Refactoring: Issue \#10 — Dead Code \texttt{GET /bid-history}}

% ----------------------------------------------------------
% ISSUE #10 — Slide 1: Problem
% ----------------------------------------------------------
\begin{frame}[fragile]{Issue \#10 — YAGNI + KISS: Dead Code Ẩn Sau \texttt{try/catch}}
  \textbf{Bối cảnh:} \texttt{GET /bid-history/:productId} là một \textbf{JSON API route}. Sau khi \texttt{try/catch} gọi \texttt{res.json()}, handler còn chứa \textbf{$\approx$30 dòng code không bao giờ chạy}.

  \bigskip
  \textbf{Nguyên tắc vi phạm:} \alert{YAGNI} + \alert{KISS}

  \bigskip
  \begin{alertblock}{Ba vấn đề cộng hưởng trong cùng dead code block}
    \begin{enumerate}
      \item \textbf{Scope error:} \texttt{productId} được khai báo bằng \texttt{const} \emph{bên trong} \texttt{try \{\}} — dead code bên ngoài reference nó → \texttt{ReferenceError} nếu được kích hoạt.
      \item \textbf{Duplicate object key:} \texttt{seller\_id} được khai báo \textbf{hai lần} trong object literal → JS silently discard key đầu tiên — latent data bug.
      \item \textbf{Double response:} \texttt{res.render()} được gọi sau \texttt{res.json()} đã sent → \texttt{"Cannot set headers after they are sent"}.
    \end{enumerate}
  \end{alertblock}

  \bigskip
  \textbf{Nguồn gốc:} Đây là phần code render-based từ version cũ của route, trước khi route được chuyển sang JSON API. Phần cũ \textbf{chưa bao giờ được xóa}.
\end{frame}

% ----------------------------------------------------------
% ISSUE #10 — Slide 2: Solution
% ----------------------------------------------------------
\begin{frame}[fragile]{Issue \#10 — Refactoring: Delete, Don't Patch}
  \textbf{Chiến lược:} \textbf{Pure deletion} — không thêm abstraction, không restructure, chỉ xóa dead code.

  \begin{columns}[t]
    \column{0.44\textwidth}
    \textbf{\alert{Trước}} — 42 dòng, 2 response path:
    \footnotesize
    \begin{verbatim}
router.get('/bid-history/:id',
  async (req, res) => {
  try {
    const productId = parseInt(...);
    const history = await ...;
    res.json({ success: true,
               data: history });
  } catch (err) { ... }

  // ← DEAD: productId out of scope
  const result = await productModel
    .findByProductId(productId);
  const product = {
    seller_id: result[0].seller_id,
    ...
    seller_id: result[0].seller_id, // dup
  }
  res.render('vwProduct/details',
             { product }); // crash
});
    \end{verbatim}

    \column{0.52\textwidth}
    \textbf{\textcolor{mygreen}{Sau}} — 9 dòng, rõ ràng:
    \footnotesize
    \begin{verbatim}
router.get('/bid-history/:id',
  async (req, res) => {
  try {
    const productId = parseInt(...);
    const history = await ...;
    res.json({ success: true,
               data: history });
  } catch (err) {
    res.status(500).json({ ... });
  }
}); // ← Clean. Done.
    \end{verbatim}

    \normalsize
    \bigskip
    \textbf{Kết quả đo lường:}
    \begin{itemize}
      \item 42 → \textbf{9 dòng} (−78\%)
      \item DB calls/request: \textbf{1 → 1} (không đổi)
      \item Latent bugs: \textbf{2 → 0}
      \item HTTP contract: \textbf{không đổi}
    \end{itemize}
  \end{columns}

  \bigskip
  \begin{center}

\begin{frame}{Issue \#5 — OCP: Thuật Toán Proxy Bidding Bị Nhúng Vào Route}
  \textbf{Vấn đề:} 4-case \texttt{if/else} cho thuật toán đấu giá tự động nằm \textbf{trực tiếp trong HTTP handler}, không được đặt tên hay cô lập.

  \bigskip
  \textbf{Nguyên tắc vi phạm:} \alert{OCP}

  \bigskip
  \begin{alertblock}{Invariants của thuật toán — không có tên, không có documentation}
    \begin{enumerate}
      \item \textbf{Self-bid:} Nếu user đã là highest bidder → chỉ update max price, không đổi current price
      \item \textbf{First bid:} Chưa có ai đặt → giá hiện tại = starting price
      \item \textbf{New bid $<$ existing max:} Người cũ thắng, giá = new bid amount
      \item \textbf{New bid $\geq$ existing max:} Người mới thắng, giá = old max + step
    \end{enumerate}
  \end{alertblock}

  \bigskip
  \textbf{Vi phạm OCP:} Thêm loại bid mới (sealed bid, Dutch auction) = \textbf{sửa trực tiếp} production route handler — không thể extend mà không modify.
\end{frame}
